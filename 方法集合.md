# 数组方法
## 在原型上添加数组去重方法
```javascript
Array.prototype.myUnique = function () {
    
    let temp = {};   //创建临时对象

    let newArr = []; //创建返回数组

    //循环数组所有元素
    for (let i = 0; i < this.length; i++) {
        //判断是不是已有属性，如果是就略过，不是就添加到对象与数组中
        if(!temp.hasOwnProperty(this[i])){
            temp[this[i]] = this[i];
            newArr.push(this[i]);
        }
    }

    return newArr;
}

let arr = [0, 0, 0, 1, 1, 2, 2, 3, 3, 'a', 'a', 'b', 'b', 'c', 'c']

console.log(arr.myUnique());
```

```javascript
Array.prototype.myUnshift = function () {
    var pos = 0;
    for (let i = 0; i < arguments.length; i++) {
        this.splice(pos, 0, arguments[i]);
        pos++;        
    }

    return  this.length
}

    var argArr = Array.prototype.slice.call(arguments);
```

# 重写typeof方法(输出所有类型)
```javascript
/*
typeof能判断的类型有string、number、boolean、undefined、function、object
*/
let myTypeof = function (val) {
    var objTypes = {
        '[object Array]': 'array',
        '[object Object]': 'object',
        '[object String]': 'string',
        '[object Math]': 'math',
        '[object Date]': 'date',
        '[object RegEx]': 'regex',
    }
    var Str = Object.prototype.toString;

    if(val === null){
        return null;
    }else if(typeof(val) === 'object'){

        let ret = Str.call(val);
        return objTypes[ret];
    }else{
        return typeof(val)
    }

}
```
